import "dart:io";
import "dart:typed_data";
import "package:flutter/material.dart";
import "package:gal/gal.dart";
import "package:video_thumbnail/video_thumbnail.dart";
import "package:path_provider/path_provider.dart";
import "package:saf_stream/saf_stream.dart";
import "permission_helper.dart";

void main() => runApp(const MyApp());

class MyApp extends StatelessWidget {
  const MyApp({super.key});
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: "Status Downloader",
      debugShowCheckedModeBanner: false,
      theme: ThemeData(
        primarySwatch: Colors.green,
        useMaterial3: true,
      ),
      home: const StatusScreen(),
    );
  }
}

class StatusScreen extends StatefulWidget {
  const StatusScreen({super.key});
  @override
  State<StatusScreen> createState() => _StatusScreenState();
}

class _StatusScreenState extends State<StatusScreen> {
  final _plugin = SafStream();
  final _helper = StatusPermissionHelper();
  List<String> _items = [];
  bool _loading = false;
  
  static const String _tree = "content://com.android.externalstorage.documents/tree/primary%3AAndroid%2Fmedia%2Fcom.whatsapp%2FWhatsApp%2FMedia%2F.Statuses";
  final String _legacyPath = "/storage/emulated/0/Android/media/com.whatsapp/WhatsApp/Media/.Statuses";

  @override
  void initState() {
    super.initState();
    _fetch();
  }

  Future<void> _fetch() async {
    setState(() => _loading = true);
    if (!(await _helper.requestStatusAccessWorkflow())) {
      _showSnack("Storage permissions are required.");
      setState(() => _loading = false);
      return;
    }
    try {
      if (Platform.isAndroid && (await _plugin.isGranted(_tree) ?? false)) {
        final files = await _plugin.files(_tree);
        setState(() {
          _items = files.map((f) => f.uri).where((u) {
            final l = u.toLowerCase();
            return l.contains(".jpg") || l.contains(".jpeg") || l.contains(".mp4");
          }).toList();
        });
        return;
      }
      final dir = Directory(_legacyPath);
      if (await dir.exists()) {
        setState(() {
          _items = dir.listSync().map((f) => f.path).where((p) {
            final l = p.toLowerCase();
            return l.endsWith(".jpg") || l.endsWith(".jpeg") || l.endsWith(".mp4");
          }).toList();
        });
      } else {
        _showSnack("WhatsApp folder not found.");
      }
    } catch (e) {
      _showSnack("Error: $e");
    } finally {
      setState(() => _loading = false);
    }
  }

  Future<void> _save(String pathOrUri) async {
    try {
      _showSnack("Saving...");
      String target = pathOrUri;
      if (pathOrUri.startsWith("content://")) {
        final temp = await getTemporaryDirectory();
        final ext = pathOrUri.toLowerCase().contains(".mp4") ? ".mp4" : ".jpg";
        final file = File("${temp.path}/status_${DateTime.now().millisecondsSinceEpoch}$ext");
        await file.writeAsBytes(await _plugin.readFileBytes(pathOrUri));
        target = file.path;
      }
      target.toLowerCase().endsWith(".mp4") ? await Gal.putVideo(target) : await Gal.putImage(target);
      _showSnack("Saved to Gallery! 🎉");
    } catch (e) {
      _showSnack("Error: $e");
    }
  }

  Future<String?> _getThumb(String pathOrUri) async {
    try {
      String target = pathOrUri;
      if (pathOrUri.startsWith("content://")) {
        final temp = await getTemporaryDirectory();
        final file = File("${temp.path}/thumb_${DateTime.now().millisecondsSinceEpoch}.mp4");
        await file.writeAsBytes(await _plugin.readFileBytes(pathOrUri));
        target = file.path;
      }
      return await VideoThumbnail.thumbnailFile(
        video: target,
        thumbnailPath: (await getTemporaryDirectory()).path,
        imageFormat: ImageFormat.JPEG,
        maxHeight: 350,
        quality: 80,
      );
    } catch (e) {
      return null;
    }
  }

  void _showSnack(String msg) {
    if (!mounted) return;
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text(msg), duration: const Duration(seconds: 2)),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("Status Downloader", style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold)),
        backgroundColor: Colors.green,
        actions: [
          IconButton(icon: const Icon(Icons.refresh, color: Colors.white), onPressed: _fetch),
        ],
      ),
      body: _loading
          ? const Center(child: CircularProgressIndicator(color: Colors.green))
          : _items.isEmpty
              ? Center(
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      const Text("No statuses found.", style: TextStyle(fontSize: 16)),
                      const SizedBox(height: 12),
                      ElevatedButton.icon(
                        onPressed: _fetch,
                        icon: const Icon(Icons.security),
                        label: const Text("Retry Check"),
                        style: ElevatedButton.styleFrom(backgroundColor: Colors.green, foregroundColor: Colors.white),
                      ),
                    ],
                  ),
                )
              : GridView.builder(
                  padding: const EdgeInsets.all(8),
                  gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
                    crossAxisCount: 2,
                    crossAxisSpacing: 8,
                    mainAxisSpacing: 8,
                    childAspectRatio: 0.85,
                  ),
                  itemCount: _items.length,
                  itemBuilder: (context, i) {
                    final item = _items[i];
                    final isVid = item.toLowerCase().contains(".mp4");
                    return Card(
                      clipBehavior: Clip.antiAlias,
                      elevation: 3,
                      child: Stack(
                        fit: StackFit.expand,
                        children: [
                          isVid
                              ? FutureBuilder<String?>(
                                  future: _getThumb(item),
                                  builder: (context, s) {
                                    if (s.connectionState == ConnectionState.done && s.data != null) {
                                      return Image.file(File(s.data!), fit: BoxFit.cover);
                                    }
                                    return const Center(child: Icon(Icons.play_circle_fill, size: 50, color: Colors.grey));
                                  },
                                )
                              : (item.startsWith("content://")
                                  ? FutureBuilder<Uint8List>(
                                      future: _plugin.readFileBytes(item),
                                      builder: (context, s) {
                                        if (s.connectionState == ConnectionState.done && s.hasData) {
                                          return Image.memory(s.data!, fit: BoxFit.cover);
                                        }
                                        return const Center(child: CircularProgressIndicator(color: Colors.green));
                                      },
                                    )
                                  : Image.file(File(item), fit: BoxFit.cover)),
                          if (isVid)
                            const Positioned(
                              top: 8,
                              right: 8,
                              child: CircleAvatar(
                                radius: 14,
                                backgroundColor: Colors.black54,
                                child: Icon(Icons.videocam, size: 16, color: Colors.white),
                              ),
                            ),
                          Positioned(
                            bottom: 0,
                            left: 0,
                            right: 0,
                            child: Container(
                              color: Colors.black54,
                              child: IconButton(
                                icon: const Icon(Icons.download, color: Colors.white),
                                onPressed: () => _save(item),
                              ),
                            ),
                          ),
                        ],
                      ),
                    );
                  },
                ),
    );
  }
}
